GET PLAYERS AND ABILITIES AND TEAMS
select 
    team.teamName,
    vbpi.firstName,
    vbpi.lastName,
    vbpi.primaryPosition,
    vbpi.pitcherRole,
    tbp.power,
    tbp.contact,
    tbp.speed,
    tbp.fielding,
    tbp.arm,
    tbp.velocity,
    tbp.junk,
    tbp.accuracy,
    tbp.age
from v_baseball_player_info vbpi
join t_baseball_players tbp
    on vbpi.baseballPlayerGUID = tbp.GUID
join t_teams team
    on tbp.teamGUID = team.GUID
